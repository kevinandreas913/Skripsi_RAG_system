How to Send API Request (JavaScript)

Berikut contoh sederhana pemanggilan API menggunakan fetch di JavaScript:


const url = "https://domainmu.com/proseschat";

const payload = {
  api_key: "TEMPEL_API_KEY_DISINI",
  message: "Kirim pertanyaan anda?"
};

fetch(url, {
  method: "POST",
  headers: {
    "Content-Type": "application/json"
  },
  body: JSON.stringify(payload)
})
.then(response => response.json())
.then(data => {
  console.log("Status:", data.status);
  console.log("Message:", data.message);
  console.log("Lengkap:", data.lengkap);
  console.log("Topk:", data.topk);
})
.catch(error => {
  console.error("Error:", error);
});
